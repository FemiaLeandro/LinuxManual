git clone <url> - clones repository

git pull origin <branch> - pull from branch

git status - shows repository data

git checkout -b <branch> - create branch

git checkout <branch> - switch to branch

git branch -D <branch> - delete branch

git checkout <file> - discard changes

git add <file/s> - add files to repo

git commit -m "<comment>" - commit changes to repo

git push origin <branch> - push changes to branch
